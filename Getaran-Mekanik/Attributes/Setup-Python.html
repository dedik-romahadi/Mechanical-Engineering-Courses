conda create -n getaran python -y
conda activate getaran

# Instal paket inti Getaran Mekanik

  jupyter numpy scipy matplotlib \
  sympy pandas control -y

  ipympl jupyterlab_widgets -y

  --user \
  --name getaran \

〰️ Cell 2 — Simulasi Getaran Bebas Teredam 
# Sistem massa-pegas-redam (underdamped)
m, c, k = 1.0, 0.5, 20.0
wn   = np.sqrt(k/m)
zeta = c / (2 * np.sqrt(k*m))

t  = np.linspace(0, 20, 1000)
y0 = [0.1, 0.0]

def damped_vib(t, y):
    return [y[1],
            -2*zeta*wn*y[1] - wn**2*y[0]]

sol = solve_ivp(damped_vib, (0,20), y0,
               t_eval=t, method='RK45')

plt.figure(figsize=(9, 3.5))
plt.plot(sol.t, sol.y[0], '#00d4ff', lw=2,
        label=f'ζ = {zeta:.3f}')
plt.axhline(0, color='gray', lw=0.8, ls='--')
plt.xlabel('Waktu (s)')
plt.ylabel('Perpindahan x(t) [m]')
plt.title(f'Getaran Bebas Teredam')
plt.legend(); plt.tight_layout()
plt.show()
