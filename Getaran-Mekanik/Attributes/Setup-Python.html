Getaran Bebas Teredam
SciPy + Matplotlib
import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

# Sistem massa-pegas-redam
m, c, k = 1.0, 0.5, 20.0
wn   = np.sqrt(k/m)
zeta = c / (2 * np.sqrt(k*m))

def vib(t, y):
    return [y[1],
            -2*zeta*wn*y[1] - wn**2*y[0]]

sol = solve_ivp(vib, (0,20),
    [0.1, 0.0], t_eval=np.linspace(0,20,500))

plt.plot(sol.t, sol.y[0], 'cyan', lw=2)
plt.title('Getaran Bebas Teredam')
plt.xlabel('Waktu (s)')
plt.ylabel('x(t) [m]')
plt.grid(); plt.show()

Respons Step (Fungsi Transfer)
python-control
import numpy as np
import matplotlib.pyplot as plt
from control import tf, step_response

# H(s) = wn²/(s²+2ζwn·s+wn²)
wn, zeta = 4.47, 0.056
num = [wn**2]
den = [1, 2*zeta*wn, wn**2]

sys = tf(num, den)
t, y = step_response(sys)

plt.plot(t, y, '#f472b6', lw=2)
plt.title('Step Response')
plt.grid(); plt.show()
