Solusi PD Simbolik
SymPy
from sympy import *
x = symbols('x')
y = Function('y')

# Selesaikan y' + 2y = 0
sol = dsolve(
  Eq(y(x).diff(x) + 2*y(x), 0)
)
print(sol)

Osilator Teredam
SciPy + Matplotlib
import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

def osc(t, y):
    return [y[1], -0.6*y[1] - 25*y[0]]

s = solve_ivp(osc, [0,10], [1,0],
    t_eval=np.linspace(0,10,400))

plt.plot(s.t, s.y[0], 'cyan')
plt.title('Osilator Teredam')
plt.grid(); plt.show()
